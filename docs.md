# TCP Multiplayer Server

## Connection

The Godot client connects to `temten.me:7777`. Configure another address in `project.godot` under `network/lobby/host` and `network/lobby/port`.

Use Node.js 20 or newer and the built-in `node:net` module. This is raw TCP, not HTTP or WebSocket.

Each packet is one UTF-8 JSON object followed by `\n`. TCP is a byte stream, so the server must buffer data until it receives a newline. Reject a line larger than 1 MiB.

## Packet Envelope

Every client request and server response uses this envelope:

```json
{
  "type": "lobby.create",
  "request_id": 2,
  "payload": {}
}
```

`request_id` is generated by the client. The server may return it to associate a response with a request.

## Required Lobby Messages

The client sends these messages:

| Type | Payload |
| --- | --- |
| `session.hello` | `{ "protocol": 1, "player_name": "Player" }` |
| `lobby.create` | `{ "max_members": 4 }` |
| `lobby.join` | `{ "lobby_id": "..." }` |
| `lobby.leave` | `{ "lobby_id": "..." }` |
| `lobby.invite` | `{ "lobby_id": "..." }` |
| `lobby.start` | `{ "lobby_id": "..." }` |

The server must return:

```json
{"type":"lobby.created","payload":{"lobby_id":"abc123","can_start":true,"players":[{"id":"p1","name":"Host"}]}}
```

Whenever membership changes, broadcast:

```json
{"type":"lobby.members","payload":{"lobby_id":"abc123","players":[{"id":"p1","name":"Host"},{"id":"p2","name":"Player 2"}]}}
```

Send failures as:

```json
{"type":"error","payload":{"code":"LOBBY_NOT_FOUND","message":"Lobby does not exist"}}
```

For `lobby.invite`, return `lobby.invite` with an `invite_code` or URL that another client can turn into `lobby.join`. For `lobby.start`, verify that the sender owns the lobby, then broadcast `lobby.started` to every member.

## Gameplay Foundation

Keep the server authoritative. Do not accept a player position or score as final state from a client.

| Direction | Type | Payload |
| --- | --- | --- |
| Client -> server | `game.input` | `{ "tick": 42, "input": { "move_x": 1, "move_y": 0, "action": false } }` |
| Server -> clients | `game.state` | `{ "tick": 42, "players": [{ "id": "p1", "x": 120, "y": 64 }] }` |
| Server -> clients | `game.event` | `{ "name": "round_finished", "winner_id": "p1" }` |

Run the simulation at a fixed tick rate, for example 20 ticks per second. Validate every input, cap message size and rate, and identify the player from the authenticated socket session rather than a player ID provided in the payload.

## Minimal Node.js Server

```js
import net from 'node:net';
import { randomUUID } from 'node:crypto';

const lobbies = new Map();

function send(socket, type, payload, requestId = null) {
  socket.write(JSON.stringify({ type, request_id: requestId, payload }) + '\n');
}

function broadcast(lobby, type, payload) {
  for (const player of lobby.players.values()) send(player.socket, type, payload);
}

function members(lobby) {
  return [...lobby.players.values()].map(({ id, name }) => ({ id, name }));
}

const server = net.createServer((socket) => {
  socket.setEncoding('utf8');
  let buffer = '';
  let player = { id: randomUUID(), name: 'Player', socket, lobbyId: null };

  socket.on('data', (chunk) => {
    buffer += chunk;
    while (buffer.includes('\n')) {
      const lineEnd = buffer.indexOf('\n');
      const line = buffer.slice(0, lineEnd);
      buffer = buffer.slice(lineEnd + 1);
      if (line.length > 1_048_576) return socket.destroy();

      let packet;
      try { packet = JSON.parse(line); } catch { send(socket, 'error', { message: 'Invalid JSON' }); continue; }
      const payload = packet.payload ?? {};

      if (packet.type === 'session.hello') {
        player.name = String(payload.player_name ?? 'Player').slice(0, 32);
        send(socket, 'session.welcome', { player_id: player.id }, packet.request_id);
      }

      if (packet.type === 'lobby.create') {
        const lobby = { id: randomUUID(), hostId: player.id, maxMembers: Math.min(Number(payload.max_members) || 4, 4), players: new Map() };
        lobby.players.set(player.id, player);
        lobbies.set(lobby.id, lobby);
        player.lobbyId = lobby.id;
        send(socket, 'lobby.created', { lobby_id: lobby.id, can_start: true, players: members(lobby) }, packet.request_id);
      }

      if (packet.type === 'lobby.join') {
        const lobby = lobbies.get(String(payload.lobby_id));
        if (!lobby) send(socket, 'error', { code: 'LOBBY_NOT_FOUND', message: 'Lobby does not exist' }, packet.request_id);
        else if (lobby.players.size >= lobby.maxMembers) send(socket, 'error', { code: 'LOBBY_FULL', message: 'Lobby is full' }, packet.request_id);
        else {
          lobby.players.set(player.id, player);
          player.lobbyId = lobby.id;
          send(socket, 'lobby.joined', { lobby_id: lobby.id, can_start: player.id === lobby.hostId, players: members(lobby) }, packet.request_id);
          broadcast(lobby, 'lobby.members', { lobby_id: lobby.id, players: members(lobby) });
        }
      }

      if (packet.type === 'lobby.invite' && player.lobbyId) {
        send(socket, 'lobby.invite', { message: 'Share this code with a player', invite_code: player.lobbyId }, packet.request_id);
      }

      if (packet.type === 'lobby.start' && player.lobbyId) {
        const lobby = lobbies.get(player.lobbyId);
        if (!lobby || lobby.hostId !== player.id) send(socket, 'error', { code: 'NOT_HOST', message: 'Only the host can start' }, packet.request_id);
        else broadcast(lobby, 'lobby.started', { lobby_id: lobby.id });
      }

      if (packet.type === 'lobby.leave' && player.lobbyId) {
        const lobby = lobbies.get(player.lobbyId);
        if (lobby) {
          lobby.players.delete(player.id);
          broadcast(lobby, 'lobby.members', { lobby_id: lobby.id, players: members(lobby) });
          if (lobby.players.size === 0) lobbies.delete(lobby.id);
        }
        player.lobbyId = null;
      }
    }
  });

  socket.on('close', () => {
    if (player.lobbyId) {
      const lobby = lobbies.get(player.lobbyId);
      if (lobby) {
        lobby.players.delete(player.id);
        broadcast(lobby, 'lobby.members', { lobby_id: lobby.id, players: members(lobby) });
        if (lobby.players.size === 0) lobbies.delete(lobby.id);
      }
    }
  });
});

server.listen(7777, '0.0.0.0');
```

The example keeps rooms in memory only. Add authentication, persistent storage, matchmaking, rate limits, TLS, and authoritative game simulation before deployment.
